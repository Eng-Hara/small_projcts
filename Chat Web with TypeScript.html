import { useState } from "react";

const DeepSeekChat = () => {
const [messages, setMessages] = useState<{ role: string; content: string }[]>([]);
const [input, setInput] = useState("");

const sendMessage = async () => {
if (!input.trim()) return;

const newMessages = [...messages, { role: "user", content: input }];
setMessages(newMessages);
setInput("");

try {
const response = await fetch("https://api.deepseek.com/chat", {
method: "POST",
headers: {
"Content-Type": "application/json",
Authorization: `Bearer YOUR_DEEPSEEK_API_KEY`,
},
body: JSON.stringify({ messages: newMessages }),
});

const data = await response.json();
if (data && data.reply) {
setMessages([...newMessages, { role: "assistant", content: data.reply }]);
}
} catch (error) {
console.error("Error fetching response:", error);
}
};

return (
<div className="flex flex-col w-full max-w-md mx-auto p-4 border rounded-lg">
<div className="h-80 overflow-y-auto border-b mb-4 p-2">
{messages.map((msg, index) => (
<div key={index} className={msg.role === "user" ? "text-right" : "text-left"}>
<p className={`p-2 rounded-lg ${msg.role === "user" ? "bg-blue-500 text-white" : "bg-gray-200"}`}>
{msg.content}
</p>
</div>
))}
</div>
<div className="flex gap-2">
<input
type="text"
value={input}
onChange={(e) => setInput(e.target.value)}
className="flex-1 p-2 border rounded"
placeholder="Type a message..."
/>
<button onClick={sendMessage} className="p-2 bg-blue-500 text-white rounded">Send</button>
</div>
</div>
);
};

export default DeepSeekChat;

I have updated the component name to "DeepSeekChat." Let me know if you need any further changes! 🚀

ChatGPT wuxuu yiri:

To run your TypeScript React chat app, follow these steps:

1. Set Up a React Project

If you don’t have a React project already, create one using Vite:

sh
npm create vite@latest my-chat-app --template react-ts
cd my-chat-app
npm install

2. Add the Chat Component

Save your DeepSeekChat.tsx file inside the src folder.

3. Use the Component

Open src/App.tsx and import your chat component:

tsx
import DeepSeekChat from "./DeepSeekChat";

function App() {
  return (
    <div className="App">
      <DeepSeekChat />
    </div>
  );
}

export default App;

4. Start the Development Server

Run:

sh
npm run dev

5. Open in Browser

After running the command, it will show a local URL (e.g., http://localhost:5173/). Open it in your browser to see your chat app.
